// ============================================================
// CONSTANT PHYSICS ENGINE
// How each constant *scales* manual adjustments
// ============================================================
function getPhysicsScale(constKey, param, rawDelta) {
  // rawDelta = how much user moved param from identity baseline
  switch(constKey) {
    case '0': return 0; // nullity — no adjustments
    case '1': return rawDelta * 1.0; // identity — unity
    case '2': return rawDelta * 2.0; // octave — double
    case '3': return rawDelta * 1.5; // triad — 3/2
    case 'π': return rawDelta * (Math.PI / 3); // cycle scaling ~1.047
    case 'φ': return rawDelta * 1.6180339887; // golden ratio
    case '√2': return rawDelta * Math.SQRT2; // ~1.414
    case 'e':  return rawDelta * (Math.E / 2); // ~1.359
    case 'i':  return rawDelta * -1; // phase inversion — perpendicular
    case 'ℙ':  {
      // prime step: which prime index based on param order
      const paramOrder = ['speed','input','output','hpass','tone','loshelf','hishelf','mix','crush','delay','feedback'];
      const idx = paramOrder.indexOf(param) % PRIMES.length;
      return rawDelta * (PRIMES[idx] / 10);
    }
    default: return rawDelta;
  }
}

// Perceptual distance of a constant from identity (0=identity, 1=max)
function getConstantDistance(constKey) {
  const distances = { '0':0.0, '1':0.0, '2':0.35, '3':0.25, 'π':0.45, 'φ':0.38, '√2':0.28, 'e':0.30, 'i':0.55, 'ℙ':0.75 };
  return distances[constKey] || 0;
}

  let target = { ...IDENTITY_STATE };

  switch(key) {
    case '0': // nullity
      target = { ...IDENTITY_STATE, output: 0.2, mix: 0, crush: 0 };
      break;
    case '1': // identity — pure reference
      target = { ...IDENTITY_STATE };
      break;
    case '2': // octave — doubled energy
      target = { ...IDENTITY_STATE, input: 1.4, output: 1.0, tone: 3, loshelf: 2, hishelf: 2 };
      break;
    case '3': // triad — harmonic richness
      target = { ...IDENTITY_STATE, loshelf: 2, hishelf: 1, mix: 0.2, mod: 'chorus' };
      break;
    case 'π': // cycle — rotating, phasing
      target = { ...IDENTITY_STATE, mix: 0.5, delay: 0.314, feedback: 0.31, mod: 'phaser' };
      break;
    case 'φ': // spiral — golden proportions
      target = { ...IDENTITY_STATE, input: 1.0, loshelf: 1.6, hishelf: 1.0, mix: 0.38, delay: 0.618, feedback: 0.38 };
      break;
    case '√2': // temperament — equal division
      target = { ...IDENTITY_STATE, hpass: 80, tone: 1.4, mix: 0.3, delay: 0.14 };
      break;
    case 'e': // decay — exponential tail
      target = { ...IDENTITY_STATE, delay: 0.5, feedback: 0.63, mix: 0.4, loshelf: 2 };

const CIRCUMFERENCE = 2 * Math.PI * 20 * (270/360);

function buildKnob(wrap) {
  const svg = document.createElementNS('http://www.w3.org/2000/svg', 'svg');
  svg.setAttribute('viewBox', '0 0 52 52');
  svg.classList.add('knob-svg');

  const cx = 26, cy = 26, r = 20;
  const trackStart = 0;
  const trackEnd = 2 * Math.PI * r * (270/360);

  const track = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
  track.setAttribute('cx', cx); track.setAttribute('cy', cy); track.setAttribute('r', r);
  track.classList.add('knob-track');
  track.style.strokeDasharray = `${trackEnd} ${2 * Math.PI * r}`;
  track.style.strokeDashoffset = 0;

  const fill = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
  fill.setAttribute('cx', cx); fill.setAttribute('cy', cy); fill.setAttribute('r', r);
  fill.classList.add('knob-fill');
  fill.style.strokeDasharray = `${trackEnd} ${2 * Math.PI * r}`;
  fill.style.strokeDashoffset = trackEnd; // empty

  const dot = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
  dot.setAttribute('cx', cx); dot.setAttribute('cy', cy + r - 5);
  dot.setAttribute('r', 2.5);
  dot.classList.add('knob-dot');

  svg.appendChild(track);
  svg.appendChild(fill);
  svg.appendChild(dot);
  wrap.appendChild(svg);
  wrap._fill = fill;
  wrap._dot = dot;
  wrap._svg = svg;

  // Mouse/touch drag
  let startY, startVal;
  const onDown = (e) => {
    e.preventDefault();
    startY = e.touches ? e.touches[0].clientY : e.clientY;
    startVal = parseFloat(wrap.dataset.value);
    document.addEventListener('mousemove', onMove);
    document.addEventListener('touchmove', onMove);
    document.addEventListener('mouseup', onUp);
    document.addEventListener('touchend', onUp);
  };
  const onMove = (e) => {
    const y = e.touches ? e.touches[0].clientY : e.clientY;
    const dy = startY - y;
    const min = parseFloat(wrap.dataset.min);
    const max = parseFloat(wrap.dataset.max);
    const range = max - min;
    const rawDelta = dy / 150 * range;

    // PHYSICS: scale delta by active constant
    const param = wrap.dataset.param;
    const scaledDelta = getPhysicsScale(activeConst, param, rawDelta);

    let newVal = Math.min(max, Math.max(min, startVal + scaledDelta));
    wrap.dataset.value = newVal;
    currentParams[param] = newVal;
    updateKnobVisual(wrap, newVal);
    updateValueDisplay(param, newVal, wrap.dataset.format);
  };
  const onUp = () => {
    document.removeEventListener('mousemove', onMove);
    document.removeEventListener('touchmove', onMove);
    document.removeEventListener('mouseup', onUp);
    document.removeEventListener('touchend', onUp);
  };
  wrap.addEventListener('mousedown', onDown);
  wrap.addEventListener('touchstart', onDown);
}

function setKnobValue(id, value) {
  const wrap = document.getElementById(id);
  if (!wrap) return;
  const min = parseFloat(wrap.dataset.min);
  const max = parseFloat(wrap.dataset.max);
  const clamp = Math.min(max, Math.max(min, value));
  wrap.dataset.value = clamp;
  currentParams[wrap.dataset.param] = clamp;
  updateKnobVisual(wrap, clamp);
  updateValueDisplay(wrap.dataset.param, clamp, wrap.dataset.format);
}

function updateKnobVisual(wrap, value) {
  if (!wrap._fill) return;
  const min = parseFloat(wrap.dataset.min);
  const max = parseFloat(wrap.dataset.max);
  const pct = (value - min) / (max - min);
  const totalArc = 2 * Math.PI * 20 * (270/360);
  const filled = pct * totalArc;
  wrap._fill.style.strokeDashoffset = totalArc - filled;

  // Rotate dot
  const angle = -135 + pct * 270;
  const rad = (angle + 135) * Math.PI / 180;
  const cx = 26, cy = 26, r = 20 - 5;
  const dx = cx + r * Math.sin(rad);
  const dy = cy - r * Math.cos(rad);
  wrap._dot.setAttribute('cx', dx);
  wrap._dot.setAttribute('cy', dy);
}

function updateValueDisplay(param, value, fmt) {
  const el = document.getElementById('val-' + param);
  if (!el) return;
  if (fmt === 'x') el.textContent = value.toFixed(2) + 'x';
  else if (fmt === 'gain') el.textContent = value.toFixed(2);
  else if (fmt === 'hz') el.textContent = Math.round(value) + 'Hz';
  else if (fmt === 'db') el.textContent = (value >= 0 ? '+' : '') + value.toFixed(1) + 'dB';
  else if (fmt === 'pct') el.textContent = Math.round(value * 100) + '%';
  else el.textContent = value.toFixed(2);
}

function onSlider(param, input) {
  const rawVal = parseFloat(input.value);
  const min = parseFloat(input.min);
  const max = parseFloat(input.max);

  // Physics scaling on delta from identity baseline
  const baseVal = IDENTITY_STATE[param] || 0;
  const rawDelta = rawVal - baseVal;
  const scaledDelta = getPhysicsScale(activeConst, param, rawDelta);
  let newVal = Math.min(max, Math.max(min, baseVal + scaledDelta));

  // If identity or nullity, just use raw
  if (activeConst === '1' || activeConst === '0') newVal = rawVal;

  currentParams[param] = newVal;
  updateSliderDisplay(param, newVal);
  updateSliderFill(input, min, max, newVal);
}

async function initAudio() {
  if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
}

async function toggleRecord() {
  await initAudio();
  if (isRecording) {
    stopRecording();
  } else {
    startRecording();
  }
}

async function startRecording() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    recordedChunks = [];
    mediaRecorder = new MediaRecorder(stream);
    mediaRecorder.ondataavailable = e => { if (e.data.size > 0) recordedChunks.push(e.data); };
    mediaRecorder.onstop = processRecording;
    mediaRecorder.start();
    isRecording = true;
    document.getElementById('btn-record').textContent = '■ stop';
    document.getElementById('rec-status').textContent = '● recording...';
  } catch(e) {
    document.getElementById('rec-status').textContent = 'mic access denied';
  }
}

function stopRecording() {
  if (mediaRecorder) mediaRecorder.stop();
  isRecording = false;
  document.getElementById('btn-record').textContent = '● record';
  document.getElementById('rec-status').textContent = 'processing...';
}

async function processRecording() {
  const blob = new Blob(recordedChunks, { type: 'audio/webm' });
  const arrayBuffer = await blob.arrayBuffer();
  sourceBuffer = await audioCtx.decodeAudioData(arrayBuffer);
  drawWaveform(sourceBuffer);
  document.getElementById('rec-status').textContent = `${sourceBuffer.duration.toFixed(1)}s recorded`;
  document.getElementById('btn-play').disabled = false;
  document.getElementById('btn-play-fx').disabled = false;
  document.getElementById('btn-export').disabled = false;
}

function playAudio() {
  if (!sourceBuffer || !audioCtx) return;
  if (currentSource) { currentSource.stop(); currentSource = null; }
  const src = audioCtx.createBufferSource();
  src.buffer = sourceBuffer;
  src.connect(audioCtx.destination);
  src.start();
  currentSource = src;
}

function playWithFX() {
  if (!sourceBuffer || !audioCtx) return;
  if (currentSource) { currentSource.stop(); currentSource = null; }

  // Build FX chain from currentParams
  const src = audioCtx.createBufferSource();
  src.buffer = sourceBuffer;
  src.playbackRate.value = currentParams.speed;

  const inputGain = audioCtx.createGain();
  inputGain.gain.value = currentParams.input;

  const outputGain = audioCtx.createGain();
  outputGain.gain.value = currentParams.output;

  const hpFilter = audioCtx.createBiquadFilter();
  hpFilter.type = 'highpass';
  hpFilter.frequency.value = currentParams.hpass;

  const toneFilter = audioCtx.createBiquadFilter();
  toneFilter.type = 'peaking';
  toneFilter.frequency.value = 1000;
  toneFilter.gain.value = currentParams.tone;

  const loShelf = audioCtx.createBiquadFilter();
  loShelf.type = 'lowshelf';
  loShelf.frequency.value = 200;
  loShelf.gain.value = currentParams.loshelf;

  const hiShelf = audioCtx.createBiquadFilter();
  hiShelf.type = 'highshelf';
  hiShelf.frequency.value = 8000;
  hiShelf.gain.value = currentParams.hishelf;

  // Delay
  const delay = audioCtx.createDelay(2.0);
  delay.delayTime.value = currentParams.delay;
  const feedback = audioCtx.createGain();
  feedback.gain.value = currentParams.feedback;
  const dryGain = audioCtx.createGain();
  dryGain.gain.value = 1 - currentParams.mix;
  const wetGain = audioCtx.createGain();
  wetGain.gain.value = currentParams.mix;

  // Bitcrusher approximation via waveshaper
  const crusher = audioCtx.createWaveShaper();
  if (currentParams.crush > 0.01) {
    const bits = Math.max(1, Math.round((1 - currentParams.crush) * 16));
    const samples = 256;
    const curve = new Float32Array(samples);
    const step = Math.pow(2, bits);
    for (let i = 0; i < samples; i++) {
      const x = (i / samples) * 2 - 1;
      curve[i] = Math.round(x * step) / step;
    }
    crusher.curve = curve;
  }

  // Chain: src → inputGain → hpFilter → toneFilter → loShelf → hiShelf → crusher → dryGain → outputGain → dest
  src.connect(inputGain);
  inputGain.connect(hpFilter);
  hpFilter.connect(toneFilter);
  toneFilter.connect(loShelf);
  loShelf.connect(hiShelf);
  hiShelf.connect(crusher);
  crusher.connect(dryGain);
  dryGain.connect(outputGain);

  // Delay loop
  crusher.connect(wetGain);
  wetGain.connect(delay);
  delay.connect(feedback);
  feedback.connect(delay);
  delay.connect(outputGain);

  outputGain.connect(audioCtx.destination);
  src.start();
  currentSource = src;
}

async function exportAudio() {
  if (!sourceBuffer) return;
  // Render offline
  const offCtx = new OfflineAudioContext(sourceBuffer.numberOfChannels, sourceBuffer.length, sourceBuffer.sampleRate);
  const src = offCtx.createBufferSource();
  src.buffer = sourceBuffer;
  src.playbackRate.value = currentParams.speed;

  const inputGain = offCtx.createGain();
  inputGain.gain.value = currentParams.input;
  const outputGain = offCtx.createGain();
  outputGain.gain.value = currentParams.output;
  const hpFilter = offCtx.createBiquadFilter();
  hpFilter.type = 'highpass';
  hpFilter.frequency.value = currentParams.hpass;
  const toneFilter = offCtx.createBiquadFilter();
  toneFilter.type = 'peaking';
  toneFilter.frequency.value = 1000;
  toneFilter.gain.value = currentParams.tone;
  const loShelf = offCtx.createBiquadFilter();
  loShelf.type = 'lowshelf';
  loShelf.frequency.value = 200;
  loShelf.gain.value = currentParams.loshelf;
  const hiShelf = offCtx.createBiquadFilter();
  hiShelf.type = 'highshelf';
  hiShelf.frequency.value = 8000;
  hiShelf.gain.value = currentParams.hishelf;

  src.connect(inputGain);
  inputGain.connect(hpFilter);
  hpFilter.connect(toneFilter);
  toneFilter.connect(loShelf);
  loShelf.connect(hiShelf);
  hiShelf.connect(outputGain);
  outputGain.connect(offCtx.destination);
  src.start();

  const rendered = await offCtx.startRendering();
  const wav = bufferToWav(rendered);
  const blob = new Blob([wav], { type: 'audio/wav' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `constants-fx-${activeConst.replace('ℙ','prime')}-${Date.now()}.wav`;
  a.click();
}

function bufferToWav(buffer) {
  const numChannels = buffer.numberOfChannels;
  const sampleRate = buffer.sampleRate;
  const format = 1;
  const bitDepth = 16;
  const blockAlign = numChannels * bitDepth / 8;
  const byteRate = sampleRate * blockAlign;
  const dataSize = buffer.length * blockAlign;
  const headerSize = 44;
  const arrayBuffer = new ArrayBuffer(headerSize + dataSize);
  const view = new DataView(arrayBuffer);

  const writeString = (offset, str) => { for (let i = 0; i < str.length; i++) view.setUint8(offset + i, str.charCodeAt(i)); };
  writeString(0, 'RIFF');
  view.setUint32(4, 36 + dataSize, true);
  writeString(8, 'WAVE');
  writeString(12, 'fmt ');
  view.setUint32(16, 16, true);
  view.setUint16(20, format, true);
  view.setUint16(22, numChannels, true);
  view.setUint32(24, sampleRate, true);
  view.setUint32(28, byteRate, true);
  view.setUint16(32, blockAlign, true);
  view.setUint16(34, bitDepth, true);
  writeString(36, 'data');
  view.setUint32(40, dataSize, true);

  let offset = 44;
  for (let i = 0; i < buffer.length; i++) {
    for (let ch = 0; ch < numChannels; ch++) {
      const sample = Math.max(-1, Math.min(1, buffer.getChannelData(ch)[i]));
      view.setInt16(offset, sample < 0 ? sample * 0x8000 : sample * 0x7FFF, true);
      offset += 2;
    }
  }
  return arrayBuffer;
}

// WAVEFORM DISPLAY
// ============================================================
function drawWaveform(buffer) {
  const canvas = document.getElementById('waveform');
  const ctx = canvas.getContext('2d');
  canvas.width = canvas.offsetWidth * window.devicePixelRatio;
  canvas.height = canvas.offsetHeight * window.devicePixelRatio;
  ctx.scale(window.devicePixelRatio, window.devicePixelRatio);

  const w = canvas.offsetWidth, h = canvas.offsetHeight;
  ctx.fillStyle = '#0a0a0a';
  ctx.fillRect(0, 0, w, h);

  const data = buffer.getChannelData(0);
  const step = Math.ceil(data.length / w);
  const accentColor = MATH_CONSTANTS[activeConst].color;

  ctx.strokeStyle = accentColor;
  ctx.lineWidth = 1;
  ctx.beginPath();

  for (let x = 0; x < w; x++) {
    let min = 1, max = -1;
    for (let j = 0; j < step; j++) {
      const d = data[(x * step) + j] || 0;
      if (d < min) min = d;
      if (d > max) max = d;
    }
    ctx.moveTo(x, (1 + min) / 2 * h);
    ctx.lineTo(x, (1 + max) / 2 * h);
  }
  ctx.stroke();
}
